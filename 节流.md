# 节流

```javascript
// Date
function throttle(func, delay) {
  let pre = 0;
  return function () {
    let context = this;
    let args = arguments;
    let now = new Date();
    if (now - pre > delay) {
      func.apply(context, args);
      pre = now;
    }
  };
}
```

```javascript
// setTimeout
function throttle1(func, delay) {
  let timer;
  return function () {
    let context = this;
    let args = arguments;
    if (timer) {
      return;
    }
    timer = setTimeout(function () {
      func.apply(context, args);
      timer = null;
    }, delay);
  };
}
```
